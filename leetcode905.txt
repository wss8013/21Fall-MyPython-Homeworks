class Solution(object):
    def sortArrayByParity(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        even_list = []
        odd_list =[]
        for i in range(len(nums)):
            if nums[i] % 2 == 0:
                even_list.append(nums[i])
            else:
                odd_list.append(nums[i])
        res = even_list + odd_list
        return res
